Les modules

=

Organisation du code

Ce sont des conteneurs pour les differentes parties de l'application (controlleurs, services, filtres, directives,...)

Exemple


  // déclaration des modules
  angular.module('myUtilityModule', []);
  // injection du module utility dans le module myAppModule
  var myAppModule = angular.module('myApp', ['myUtilityModule']);

  // ajout d'un filtre au module
  myAppModule.filter('greet', function() {
    return function(name) {
      return 'Hello, ' + name + '!';
    };
  });

  myAppModule.controller(...);
  myAppModule.directive(...);
  

<div ng-app="myApp"> <!-- intitalisation du module myApp -->
  <div>
    {{ 'World' | greet }}
  </div>
</div>
