{{ 'World' | greet }}

app.js


// d√©claration du module
var myAppModule = angular.module('myApp', []);

// ajout d'un filtre au module
myAppModule.filter('greet', function() {
  return function(name) {
    return 'Hello, ' + name + '!';
  };
});

index.html


<div ng-app="myApp"> <!-- intitalisation du module myApp -->
  <div>
    {{ 'World' | greet }}
  </div>
</div>
