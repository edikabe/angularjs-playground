Controller & Scope Notre premier contrôleur

app.js


          var app = angular.module('myApp', []);

          app.controller('TodoCtrl', ['$scope', function($scope) {
              $scope.todos = [{
                  completed: true,
                  title: 'Finaliser la présentation'
              }, {
                  completed: false,
                  title: 'Organiser le quickie'
              }];

              $scope.addTodo = function() {
                  var newTodo = {
                      title: $scope.newTodo.trim(),
                      completed: false
                  };

                  $scope.todos.push(newTodo);
              };
          }]);

index.html


          <html ng-app="myApp">
            <head>
              <script src="http://ajax.googleapis.com/ajax/libs/angularjs/1.3.12/angular.min.js"></script>
              <script src="/demos/03-todotask/app.js"></script>
            </head>
            <body>
                 <form ng-submit="addTodo()">
                        <input ng-model="newTodo" />
                        <button type="submit">Ajouter</button>
                </form>
                <ul>
                    <li ng-repeat="todo in todos">
                        <input type="checkbox" ng-model="todo.completed">
                        <span>{{todo.title}}</span>
                    </li>
                </ul>
            </body>
          </html>
